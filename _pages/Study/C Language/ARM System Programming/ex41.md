int Uart_Get_Int_Num(void)
{

    char s[10];
    int i;
	int num = 0;
	
    for (i=0;i<4;i++)
    {
        
        while (!Macro_Check_Bit_Set(USART1->SR, 5));
        s[i] = USART1->DR;

        while (!Macro_Check_Bit_Set(USART1->SR, 7));
        USART1->DR = s[i];

        if (s[i] == '\r')
        {
            return num;
        }

		num = num * 10 + (s[i] - '0');
	}
}
