# 정답 코드
---

```c
void Main(void)
{
   Sys_Init();

   char x;
   int n;
   int timer_run = 0;
   int led = 0;
   TIM4_Repeat(10);
   TIM2_Repeat(100);

   for(;;)
   {
      x = Uart1_Get_Pressed();
      if(x != 0) 
      {
         n = x - '0';
         Uart1_Printf("%d", n);

         // 0? timer_run = 0;
         if (n==0)
         {
            timer_run = 0;
         }
         // 1~9? TIM4_Change_Value(n * 10); timer_run = 1
         else
         {
            TIM4_Change_Value(n * 10); 
            timer_run = 1;
         }
      }

      if(TIM4_Check_Timeout() && timer_run == 1)
      {
         //LED반전
         LED_Display(led ^= 1);
      }
      
      if(TIM2_Check_Timeout())
      {
         Uart1_Printf(".");
      }
      
   }
}
```
